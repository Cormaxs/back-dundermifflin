function listFilesFromMultipleUrls() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var ui = SpreadsheetApp.getUi();
  
  // Pedir los enlaces al usuario
  var response = ui.prompt('Introduce los enlaces de las carpetas separados por comas:');
  var input = response.getResponseText();
  
  if (!input) return; // Si cancela o está vacío, no hace nada

  // Separar el texto por comas y limpiar espacios en blanco
  var urls = input.split(',').map(function(item) { return item.trim(); });
  
  var fileData = [];
  fileData.push(["Nombre del Archivo", "Enlace URL", "Carpeta Origen"]);

  var errors = [];

  urls.forEach(function(url) {
    try {
      var folderId = extractIdFromUrl(url);
      if (folderId) {
        var folder = DriveApp.getFolderById(folderId);
        // Pasamos también el nombre de la carpeta raíz para saber de dónde viene cada archivo
        processFolder(folder, fileData);
      } else {
        errors.push("URL no válida: " + url);
      }
    } catch (e) {
      errors.push("Error con la carpeta (" + url + "): " + e.toString());
    }
  });

  // Escribir los resultados
  if (fileData.length > 1) {
    sheet.clear();
    sheet.getRange(1, 1, fileData.length, 2).setValues(fileData.map(function(row) {
      return [row[0], row[1]]; // Solo tomamos nombre y URL para la hoja
    }));
    sheet.getRange(1, 1, 1, 2).setFontWeight("bold");
    
    var msg = 'Se encontraron ' + (fileData.length - 1) + ' archivos en total.';
    if (errors.length > 0) msg += '\n\nHubo errores en algunos enlaces:\n' + errors.join('\n');
    ui.alert(msg);
  } else {
    ui.alert('No se encontraron archivos o los enlaces no eran válidos.');
  }
}

/**
 * Función auxiliar para extraer el ID
 */
function extractIdFromUrl(url) {
  var match = url.match(/[-\w]{25,}/);
  return match ? match[0] : null;
}

/**
 * Función recursiva para procesar carpetas
 */
function processFolder(folder, fileData) {
  var files = folder.getFiles();
  while (files.hasNext()) {
    var file = files.next();
    fileData.push([file.getName(), file.getUrl()]);
  }
  
  var subfolders = folder.getFolders();
  while (subfolders.hasNext()) {
    var subfolder = subfolders.next();
    processFolder(subfolder, fileData);
  }
}
